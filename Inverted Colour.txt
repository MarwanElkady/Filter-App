#include <iostream>
#include "Image_Class.h"
using namespace std;

int main (){
    Image image("mario.jpg");


    for (int i = 0; i < image.width; ++i){
        for (int j = 0; j < image.height; ++j){
            for (int k = 0; k < image.channels; ++k){
                image(i,j,k) = 255 - image(i,j,k);
            }
        }
    }

    image.saveImage("MarioInverted.png"); 
    
    return 0;
}
